Looking at your code, the issue where messages appear twice is likely caused by **multiple WebSocket connections**. Here's what's happening and how to fix it:

## The Problem

When you click "Connect" multiple times, you're creating multiple WebSocket connections to the same server. Each connection will receive and display the same message, causing duplicates.

## The Solution

Add a check to prevent multiple connections and properly clean up any existing connection before creating a new one:

```javascript
let socket = null;

// ... (rest of your existing code)

function connect() {
  // Prevent multiple connections
  if (socket && socket.readyState === WebSocket.OPEN) {
    alert('Already connected! Please disconnect first.');
    return;
  }

  // Close any existing connection before creating a new one
  if (socket) {
    socket.close();
    socket = null;
  }

  const baseUrl = serverUrlInput.value.trim();
  const name = nameInput.value.trim();

  if (!baseUrl || !name) {
    alert('Please enter server URL and name');
    return;
  }

  const url = baseUrl + '?name=' + encodeURIComponent(name);

  try {
    socket = new WebSocket(url);

    socket.onopen = () => {
      setStatus('Connected');
      connectBtn.disabled = true;
      disconnectBtn.disabled = false;
    };

    socket.onclose = () => {
      setStatus('Disconnected');
      connectBtn.disabled = false;
      disconnectBtn.disabled = true;
      socket = null;
    };

    socket.onerror = (error) => {
      console.error('WebSocket error:', error);
      setStatus('Error - Disconnected');
    };

    socket.onmessage = (event) => {
      try {
        const data = JSON.parse(event.data);
        const time = new Date(data.timestamp || Date.now()).toLocaleTimeString();

        if (data.type === 'message') {
          addMessage(`[${time}] ${data.name}: ${data.text}`);
        } else if (data.type === 'join') {
          addMessage(`[${time}] ${data.name} joined`);
        } else if (data.type === 'leave') {
          addMessage(`[${time}] ${data.name} left`);
        } else if (data.type === 'error') {
          addMessage(`[ERROR] ${data.message}`);
        }
      } catch (e) {
        console.error('Error parsing message:', e);
        addMessage('Invalid message received');
      }
    };
  } catch (error) {
    console.error('Connection error:', error);
    setStatus('Connection failed');
    connectBtn.disabled = false;
    disconnectBtn.disabled = true;
    socket = null;
  }
}
```
