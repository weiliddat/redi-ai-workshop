  if (!socket || socket.readyState !== WebSocket.OPEN) {
    return;
  }

  const text = messageInput.value.trim();
  if (!text) return;

    const newName = text.substring(6).trim(); // Remove "/nick " part

        type: 'nick',
        name: newName

    }

    return; // Don't send as regular message
  }

  // Regular message
  socket.send(JSON.stringify({
    type: 'message',
    text: text
  }));

  messageInput.value = '';
});

### 4. **Handle `nick` broadcast messages**
Add a case in your `socket.onmessage` handler:
```javascript
socket.onmessage = (event) => {
  try {
    const data = JSON.parse(event.data);
    const time = new Date(data.timestamp || Date.now()).toLocaleTimeString();

    if (data.type === 'message') {
      addMessage(`[${time}] ${data.name}: ${data.text}`);
    } else if (data.type === 'join') {
      addMessage(`[${time}] ${data.name} joined`);
    } else if (data.type === 'leave') {
      addMessage(`[${time}] ${data.name} left`);
    } else if (data.type === 'nick') {
      // Update our tracked name if this change is for us
      if (data.oldName === currentName) {
        currentName = data.name;
        nameInput.value = currentName; // Update the name field
      }
      addMessage(`[${time}] ${data.oldName} changed their name to ${data.name}`);
    } else if (data.type === 'list') {
      // Handle list response (optional UI improvement)
      const names = data.names.join(', ');
      addMessage(`[${time}] Connected users: ${names}`);
    } else if (data.type === 'error') {
      addMessage(`[ERROR] ${data.message}`);
    }
  } catch (e) {
    addMessage('Invalid message received');
  }
};
```

## What Happens When You Use `/nick`

1. **You type**: `/nick NewName` and press Enter
2. **Client sends**: `{"type": "nick", "name": "NewName"}`
3. **Server validates**:
   - Checks if "NewName" is available
   - If available: Updates your name server-side and broadcasts to everyone
   - If taken: Sends error back to you only
4. **You receive broadcast**: `{"type": "nick", "oldName": "OldName", "name": "NewName"}`
5. **Your UI updates**:
   - Chat shows: `[time] OldName changed their name to NewName`
   - Name input field updates to "NewName" (so future `/nick` commands reference your current name)
   - Future messages you send will appear as "NewName"
