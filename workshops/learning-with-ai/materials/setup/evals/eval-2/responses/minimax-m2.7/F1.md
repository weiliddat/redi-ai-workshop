### Option 2: Custom Format (24-hour with seconds)

Add a helper function before `setStatus`:

```javascript
function formatTimestamp(timestamp) {
  const date = new Date(timestamp || Date.now());
  const pad = (n) => n.toString().padStart(2, '0');
  const hours = pad(date.getHours());
  const minutes = pad(date.getMinutes());
  const seconds = pad(date.getSeconds());
  const day = pad(date.getDate());
  const month = pad(date.getMonth() + 1);
  const year = date.getFullYear();
  return `${day}/${month}/${year} ${hours}:${minutes}:${seconds}`;
}
```

**Then change the onmessage handler:**
```javascript
const time = formatTimestamp(data.timestamp);
```

**Result:** `[15/06/2025 14:30:00] Alice: Hello!`

### Option 4: Add Timestamps to All Message Types

Your code handles `join`, `leave`, `nick`, `list`, and `error` messages, but let's ensure all have consistent formatting:

**Replace your entire `onmessage` handler:**

```javascript
socket.onmessage = (event) => {
  try {
    const data = JSON.parse(event.data);
    const time = new Date(data.timestamp || Date.now()).toLocaleString();

    switch (data.type) {
      case 'message':
        addMessage(`[${time}] ${data.name}: ${data.text}`);
        break;
      case 'join':
        addMessage(`[${time}] ★ ${data.name} joined the chat`);
        break;
      case 'leave':
        addMessage(`[${time}] ☆ ${data.name} left the chat`);
        break;
      case 'nick':
        addMessage(`[${time}] ↪ ${data.oldName} changed name to ${data.name}`);
        break;
      case 'list':
        addMessage(`[${time}] Users online: ${data.names.join(', ')}`);
        break;
      case 'error':
        addMessage(`[${time}] ❌ ERROR: ${data.message}`);
        break;
      default:
        addMessage(`[${time}] ${event.data}`);
    }
  } catch (e) {
    addMessage(`[${new Date().toLocaleString()}] Invalid message received`);
  }
};
```
