  e.preventDefault();

  if (!socket || socket.readyState !== WebSocket.OPEN) {
    return;
  }

  const text = messageInput.value.trim();
  if (!text) return;

  // --- BEGIN NEW LOGIC ---
  if (text.startsWith('/nick ')) {
    const newName = text.substring(6).trim(); // Get everything after "/nick "

    type: 'message',
    text: text

});
