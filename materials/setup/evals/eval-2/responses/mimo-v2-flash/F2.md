## 3. **Update JavaScript to Handle Name Changes**

Here's the updated JavaScript with full `/nick` command support:

```javascript
let socket = null;
let currentName = ''; // Track the current name

const serverUrlInput = document.getElementById('serverUrl');
const nameInput = document.getElementById('nameInput');
const connectBtn = document.getElementById('connectBtn');
const disconnectBtn = document.getElementById('disconnectBtn');
const statusEl = document.getElementById('status');
const messagesEl = document.getElementById('messages');
const chatForm = document.getElementById('chatForm');
const messageInput = document.getElementById('messageInput');
const currentNameEl = document.getElementById('currentName');

function setStatus(text) {
  statusEl.textContent = text;
}

function updateCurrentName(name) {
  currentName = name;
  currentNameEl.textContent = name;
}

function addMessage(text) {
  const div = document.createElement('div');
  div.className = 'message';
  div.textContent = text;
  messagesEl.appendChild(div);
  messagesEl.scrollTop = messagesEl.scrollHeight;
}

function connect() {
  const baseUrl = serverUrlInput.value.trim();
  const name = nameInput.value.trim();

  if (!baseUrl || !name) {
    alert('Please enter server URL and name');
    return;
  }

  const url = baseUrl + '?name=' + encodeURIComponent(name);

  socket = new WebSocket(url);

  socket.onopen = () => {
    setStatus('Connected');
    connectBtn.disabled = true;
    disconnectBtn.disabled = false;
    updateCurrentName(name);
    addMessage(`[System] Connected as ${name}`);
  };

  socket.onclose = () => {
    setStatus('Disconnected');
    connectBtn.disabled = false;
    disconnectBtn.disabled = true;
    updateCurrentName('Not connected');
    socket = null;
  };

  socket.onerror = () => {
    setStatus('Disconnected');
  };

  socket.onmessage = (event) => {
    try {
      const data = JSON.parse(event.data);
      const time = new Date(data.timestamp || Date.now()).toLocaleTimeString();

      if (data.type === 'message') {
        addMessage(`[${time}] ${data.name}: ${data.text}`);
      } else if (data.type === 'join') {
        addMessage(`[${time}] ${data.name} joined`);
      } else if (data.type === 'leave') {
        addMessage(`[${time}] ${data.name} left`);
      } else if (data.type === 'error') {
        addMessage(`[ERROR] ${data.message}`);
      } else if (data.type === 'nick') {
        // Handle name change
        if (data.oldName === currentName) {
          // This is our own name change
          updateCurrentName(data.name);
          addMessage(`[${time}] You changed your name to ${data.name}`);
        } else {
          // Someone else changed their name
          addMessage(`[${time}] ${data.oldName} changed their name to ${data.name}`);
        }
      } else if (data.type === 'list') {
        // Handle /list command response
        addMessage(`[${time}] Users online: ${data.names.join(', ')}`);
      }
    } catch (e) {
      addMessage('Invalid message received');
    }
  };
}

function disconnect() {
  if (socket) {
    socket.close();
  }
}

// Helper function to handle commands
function handleCommand(text) {
  const parts = text.split(' ');
  const command = parts[0].toLowerCase();
  
  if (command === '/nick') {
    if (parts.length < 2) {
      addMessage('[ERROR] Usage: /nick <new-name>');
      return true;
    }
    
    const newName = parts[1].trim();
    if (!newName) {
      addMessage('[ERROR] Name cannot be empty');
      return true;
    }
    
    // Send nick command to server
    socket.send(JSON.stringify({
      type: 'nick',
      name: newName
    }));
    
    return true; // Command handled
  }
  
  if (command === '/list') {
    // Send list command to server
    socket.send(JSON.stringify({
      type: 'list'
    }));
    
    return true; // Command handled
  }
  
  return false; // Not a command
}

connectBtn.addEventListener('click', connect);
disconnectBtn.addEventListener('click', disconnect);

chatForm.addEventListener('submit', (e) => {
  e.preventDefault();

  if (!socket || socket.readyState !== WebSocket.OPEN) {
    return;
  }

  const text = messageInput.value.trim();
  if (!text) return;

  // Check if it's a command
  if (handleCommand(text)) {
    messageInput.value = '';
    return;
  }

  // Regular message
  socket.send(JSON.stringify({
    type: 'message',
    text: text
  }));

  messageInput.value = '';
});

// Also allow changing name by pressing Enter in the name input field
nameInput.addEventListener('keypress', (e) => {
  if (e.key === 'Enter' && socket && socket.readyState === WebSocket.OPEN) {
    const newName = nameInput.value.trim();
    if (newName && newName !== currentName) {
      socket.send(JSON.stringify({
        type: 'nick',
        name: newName
      }));
    }
  }
});
```

## 4. **Key Changes Explained**

### **State Management**
- Added `currentName` variable to track the current display name
- Added `updateCurrentName()` function to update both the variable and UI

### **Command Handling**
- Added `handleCommand()` function that detects commands starting with `/`
- `/nick <new-name>` sends a "nick" type message to the server
- `/list` sends a "list" type message to the server

### **Server Response Handling**
- Added handling for `nick` type messages:
  - If it's your own name change (oldName matches currentName), update the current name
  - If it's someone else's name change, display it normally
- Added handling for `list` type messages to display the user list

### **UX Improvements**
- Shows current name in the UI
- Allows changing name by pressing Enter in the name input field
- Better error messages for commands
- System messages for connection and name changes

## 5. **Usage Examples**

Once connected, users can:
- Change their name: `/nick Alice`
- List users: `/list`
- Change name by typing a new name in the name field and pressing Enter

Make sure your WebSocket server handles these message types:

- `"type": "list"` - should respond with `"type": "list"` and `"names"` array
- Broadcast `"type": "nick"` messages when names change
- Send errors when names are already taken
