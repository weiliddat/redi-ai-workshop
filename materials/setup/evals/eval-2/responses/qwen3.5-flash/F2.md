2.  **Command Parsing:** Intercept the chat input to detect if the user typed `/nick`.
3.  **Message Handling:** Update the `onmessage` handler to display name change broadcasts from the server.

Here is the walkthrough of the changes and the updated code.

### The Modified Code

Here is the updated `<script>` section with the necessary logic. You can replace the existing `<script>` tag in your HTML with this version.

```html
  <script>
    let socket = null;
    let myName = ''; // State to track current local name

    const serverUrlInput = document.getElementById('serverUrl');
    const nameInput = document.getElementById('nameInput');
    const connectBtn = document.getElementById('connectBtn');
    const disconnectBtn = document.getElementById('disconnectBtn');
    const statusEl = document.getElementById('status');
    const messagesEl = document.getElementById('messages');
    const chatForm = document.getElementById('chatForm');
    const messageInput = document.getElementById('messageInput');

    function setStatus(text) {
      statusEl.textContent = text;
    }

    function addMessage(text) {
      const div = document.createElement('div');
      div.className = 'message';
      // Check if text starts with [ERROR] to style it differently
      if (text.startsWith('[ERROR]')) {
        div.style.color = 'red';
      }
      div.textContent = text;
      messagesEl.appendChild(div);
      messagesEl.scrollTop = messagesEl.scrollHeight;
    }

    function connect() {
      const baseUrl = serverUrlInput.value.trim();
      const name = nameInput.value.trim();

      if (!baseUrl || !name) {
        alert('Please enter server URL and name');
        return;
      }

      const url = baseUrl + '?name=' + encodeURIComponent(name);

      socket = new WebSocket(url);

      socket.onopen = () => {
        // Store the connected name locally
        myName = name;
        nameInput.value = name; // Update the input UI to match
        setStatus('Connected (Display Name: ' + name + ')');
        connectBtn.disabled = true;
        disconnectBtn.disabled = false;
      };

      socket.onclose = () => {
        setStatus('Disconnected');
        connectBtn.disabled = false;
        disconnectBtn.disabled = true;
        socket = null;
        myName = '';
      };

      socket.onerror = () => {
        setStatus('Disconnected');
      };

      socket.onmessage = (event) => {
        try {
          const data = JSON.parse(event.data);
          const time = new Date(data.timestamp || Date.now()).toLocaleTimeString();

          // Handle Standard Chat Messages
          if (data.type === 'message') {
            addMessage(`[${time}] ${data.name}: ${data.text}`);
          } 
          // Handle Join
          else if (data.type === 'join') {
            addMessage(`[${time}] ${data.name} joined`);
          } 
          // Handle Leave
          else if (data.type === 'leave') {
            addMessage(`[${time}] ${data.name} left`);
          } 
          // Handle Nickname Updates
          else if (data.type === 'nick') {
            // Update local UI if it's your name changing
            if (data.oldName === myName) {
              myName = data.name;
              nameInput.value = data.name;
              setStatus('Connected (Display Name: ' + data.name + ')');
            }
            // Log the change for everyone to see
            addMessage(`[${time}] ${data.oldName} changed name to ${data.name}`);
          } 
          // Handle Errors
          else if (data.type === 'error') {
            addMessage(`[ERROR] ${data.message}`);
          }
        } catch (e) {
          addMessage('Invalid message received');
        }
      };
    }

    function disconnect() {
      if (socket) {
        socket.close();
      }
    }

    connectBtn.addEventListener('click', connect);
    disconnectBtn.addEventListener('click', disconnect);

    chatForm.addEventListener('submit', (e) => {
      e.preventDefault();

      if (!socket || socket.readyState !== WebSocket.OPEN) {
        return;
      }

      let rawText = messageInput.value.trim();
      if (!rawText) return;

      // Check for /nick command
      if (rawText.startsWith('/nick')) {
        const parts = rawText.split(' ');
        const newName = parts.slice(1).join(' ').trim();

        if (!newName) {
          addMessage('[ERROR] Please provide a new name (e.g., /nick NewName)');
          return;
        }

        // Validate length
        if (newName.length > 20) {
             addMessage('[ERROR] Name too long');
             return;
        }

        // Send nickname change command
        socket.send(JSON.stringify({
          type: 'nick',
          name: newName
        }));

        messageInput.value = '';
        return; // Stop further processing
      }

      // Send Standard Message
      socket.send(JSON.stringify({
        type: 'message',
        text: rawText
      }));

      messageInput.value = '';
    });
  </script>
```

### Key Details on Implementation

1.  **Local Validation (`parts.slice(1)`):**
    When checking for `/nick`, splitting the string allows you to handle multi-word names (like `/nick Captain America`). `slice(1)` ensures you don't just get "Captain" if there are spaces.

2.  **State Synchronization (`nameInput.value`):**
    Updating the HTML `<input id="nameInput">` when the `nick` command succeeds ensures that if the user disconnects and clicks "Connect" again, the form is pre-filled with their new name.

3.  **Broadcast Logic:**
    The server broadcasts `{ oldName, name }` to everyone. We check `if (data.oldName === myName)` to update the user interface. If the name changed to match someone else's old name (e.g., you try to take "Alice"'s name), you would see the broadcast, but your local `myName` variable wouldn't trigger an update if you weren't the one who changed it (though this specific logic depends on server-side validation).

4.  **Error Display:**
    The spec specifies that if the name is taken, the server returns `{ type: 'error', message: ... }`. This is handled in your existing `else if (data.type === 'error')` block, ensuring the user gets immediate feedback.
